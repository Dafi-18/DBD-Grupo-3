CREATE TABLE IF NOT EXISTS Venta( 
		Id_venta text PRIMARY KEY,
		Id_usuario text,
 		Fecha_venta date,
 		Monto_final decimal,
 		Estado_pago text
);

CREATE TABLE IF NOT EXISTS Detalle_venta (
    Id_venta text,
    Id_articulo text,
    Medio_pago text,
    cantidad INT,
    subtotal DECIMAL,
    PRIMARY KEY (Id_venta, Id_articulo)
);
